function login() { const emailInput = document.getElementById('email').value; const passInput = document.getElementById('pass').value; // جلب البيانات التي خزنها المستخدم عند التسجيل const savedData = localStorage.getItem('registeredUser'); if (savedData) { const user = JSON.parse(savedData); // التأكد من تطابق الإيميل والباسورد if (emailInput === user.email && passInput === user.password) { alert("تم تسجيل الدخول بنجاح!"); // إضافة كود عشوائي (?v=) لإجبار المتصفح على تحميل النسخة الجديدة من الشكل const version = new Date().getTime(); window.location.href = 'index.html?v=' + version; } else { alert("خطأ في الإيميل أو كلمة المرور!"); } } else { alert("لا يوجد حساب مسجل بهذا الإيميل، يرجى إنشاء حساب أولاً."); } }
